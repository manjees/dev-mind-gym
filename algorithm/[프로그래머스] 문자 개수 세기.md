fun solution(my_string: String): IntArray {
  val answer = IntArray(52)

  for (char in my_string) {
    when {
      char in 'A'..'Z' -> answer[char - 'A']++
      char in 'a'..'z' -> answer[char - 'a' + 26]++
    }
  }
  return answer
}
