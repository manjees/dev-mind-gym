fun solution(arr: IntArray, queries: Array<IntArray>): IntArray {
  queries.forEachIndexed { index, ints ->
    (ints.first() .. ints[1]).forEach { i ->
      if (i % ints.last()  == 0) {
        arr[i] += 1
      }
    }
  }
  return arr
}
