class Solution {
    fun solution(new_id: String): String {
        var id = new_id

        id = id.lowercase()

        id = id.replace("[^a-z0-9-_.]".toRegex(), "")

        id = id.replace("[.]{2,}".toRegex(), ".")

        id = id.trim('.')

        if (id.isEmpty()) id = "a"

        if (id.length >= 16) {
            id = id.substring(0, 15)
            id = id.trimEnd('.')
        }

        while (id.length < 3) {
            id += id.last()
        }

        return id
    }
}
