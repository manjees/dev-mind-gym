fun solution(l: Int, r: Int): IntArray {
        val resultList = arrayListOf<Int>()
        for (i in l..r) {
            if (i % 5 == 0) {
                var isAnswer = true
                i.toString().forEach {
                    if (it != '5' && it != '0') {
                        isAnswer = false
                    }
                }
                if (isAnswer) resultList.add(i)
            }
        }
        return if (resultList.isEmpty()) {
            intArrayOf(-1)
        } else {
            resultList.toIntArray()
        } 
    }
