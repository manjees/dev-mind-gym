fun solution(intStrs: Array<String>, k: Int, s: Int, l: Int): IntArray {
    var answer = arrayListOf<Int>()

    intStrs.forEach {
        val num = it.substring(s..s + l - 1).toInt()
        if (num > k) answer.add(num)
    }
    return answer.toIntArray()
}
