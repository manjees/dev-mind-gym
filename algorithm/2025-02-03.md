```kotlin
// 2와 5로 나누어 떨어지지 않는 정수 n(1 ≤ n ≤ 10000)가 주어졌을 때, 각 자릿수가 모두 1로만 이루어진 n의 배수를 찾는 프로그램을 작성하시오.

fun test(n: Int) {
        var num = 0

        for (i in 1..Int.MAX_VALUE) {
            num = num * 10 + 1
            num %= n

            if (num == 0) {
                println(i)
                break
            }
        }
    }
```
